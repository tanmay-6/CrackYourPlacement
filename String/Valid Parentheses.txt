class Solution {
public:
    bool isValid(string s) {
        stack<char> st;
        for(auto& ch : s){
            if(ch=='(' || ch=='{' || ch=='['){
                st.push(ch);
            }
            else if(!st.empty()){
                if(st.top()=='(' && ch==')') st.pop();
                else if(st.top()=='[' && ch==']') st.pop();
                else if(st.top()=='{' && ch=='}') st.pop();
                else{
                    st.push(ch);
                }
            }
            else{
                st.push(ch);
            }
        }
        return st.empty();
    }
};
