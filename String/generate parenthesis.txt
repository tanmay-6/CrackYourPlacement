class Solution {
public:
    void solve(string s, int rem, int clc, vector<string>& ans){
        if(rem == 0 && clc == 0){
            ans.push_back(s);
            return;
        }

        if(rem > 0){
            solve(s+'(', rem-1, clc+1, ans);
        }
        if(clc > 0){
            solve(s+')', rem, clc-1, ans);
        }
    }
    vector<string> generateParenthesis(int n) {
        vector<string> ans;
        string s= "";
        solve(s,n,0, ans);
        return ans;
    }
};
