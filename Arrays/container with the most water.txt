class Solution {
public:
    int maxArea(vector<int>& h) {
        int n = h.size();
        int i = 0;
        int j = n-1;
        int water = INT_MIN;
        while(i<=j){
            water = max(water, (j-i)*(min(h[i],h[j])));
            if(h[i] < h[j]){
                i++;
            }
            else{
                j--;
            }
        }
        return water;
    }
};
//easy two pointer solution
