class Solution {
public:
    int maxProfit(vector<int>& prices) {
        vector<int> mini;
        int minimum = INT_MAX;
        for(int i=0; i<prices.size(); i++){
            minimum = min(minimum, prices[i]);
            mini.push_back(minimum);
        }
        int ans = 0;
        for(int i=0 ; i<prices.size(); i++){
            ans = max(ans,prices[i]-mini[i]);
        }
        return ans;
    }
};
