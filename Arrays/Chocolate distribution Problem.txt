class Solution{
    public:
    long long min(long long a, long long b){
        if(a<b) return a;
        return b;
    }
    long long findMinDiff(vector<long long> a, long long n, long long m){
        sort(a.begin(),a.end());
        int ans = INT_MAX;
        for(long long i=0; i<=n-m; i++){
            ans = min(ans, a[i+m-1]-a[i]);
        }
        return ans;
    
    }   
};
