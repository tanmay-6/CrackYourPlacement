class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        vector<int> ans;
        for(int i=0; i<nums.size(); i++){
            if(nums[i]!=0){
                ans.push_back(nums[i]);
                nums[i]=0;
            }
        }
        for(int j=0; j<ans.size(); j++){
            nums[j] = ans[j];
        }
    }
};
